\newcommand{\vsm}[1]{\vc{\sm{#1}}} % <> as delimiters

% calculus
\newcommand{\p}{\partial}
% usage: \dv[order of derivative]{func at top}{respect to}
% \dv{x} => d/dx
% \dv{y}{x} => dy/dx
% \dv[2]{x} => d^2/dx^2
% \dv[3]{y}{x} => d^3y/dx^3

\NewDocumentCommand{\dv}{ O{} m g }{
    \IfNoValueTF{#3}{ % no value of #3
        \frac{d^{#1}}{d #2^{#1}}
    }{ % value of #3
        \frac{d^{#1} #2}{d #3^{#1}}
    }
}
\NewDocumentCommand{\pdv}{ O{} m g }{
    \IfNoValueTF{#3}{ % no value of #3
        \frac{\p^{#1}}{\p #2^{#1}}
    }{ % value of #3
        \frac{\p^{#1} #2}{\p #3^{#1}}
    }
}
